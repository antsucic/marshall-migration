INSERT INTO staging.companies
(
    legacy_id
    , legacy_source
    , legacy_user_id
    , legacy_location_id
    , name
    , phone
    , status
)
SELECT
    owners."Company_Id"
    , '#{source}'
    , entities."Id"
    , entities."Location_Id"
    , owners."Display_Name"
    , owners."Work_Phone"
    , owners."Status"
FROM
    "#{source}"."Owners" owners
    LEFT JOIN "#{source}"."Entities" entities
        ON owners."Email_Address" = entities."Login_Name"
;
